create or replace TABLE DIM_CUSTOMER (
	CUSTOMER_ID NUMBER(38,0) NOT NULL autoincrement start 1 increment 1 order,
	CUSTOMER VARCHAR(20),
	SEARCH_NAME VARCHAR(50),
	COMPANY VARCHAR(50),
	CREATED_DT TIMESTAMP_NTZ(0),
	CREATED_BY VARCHAR(50),
	MODIFIED_DT TIMESTAMP_NTZ(0),
	MODIFIED_BY VARCHAR(50),
	ACTIVE_STATUS VARCHAR(1),
	PARENTCLIENTNUMBER VARCHAR(36),
	ACCOUNT_TYPE VARCHAR(2000)
);
