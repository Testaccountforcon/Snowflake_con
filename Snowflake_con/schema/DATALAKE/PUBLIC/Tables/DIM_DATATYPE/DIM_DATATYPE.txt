create or replace TABLE DIM_DATATYPE (
	DATATYPE_ID NUMBER(38,0) NOT NULL autoincrement start 1 increment 1 order,
	DATATYPE_VAL VARCHAR(50),
	DATATYPE_DESC VARCHAR(50),
	VALID_FRM_DT TIMESTAMP_NTZ(0),
	VALID_TO_DT TIMESTAMP_NTZ(0),
	CREATED_DT TIMESTAMP_NTZ(0),
	CREATED_BY VARCHAR(50),
	MODIFIED_DT TIMESTAMP_NTZ(0),
	MODIFIED_BY VARCHAR(50)
);
