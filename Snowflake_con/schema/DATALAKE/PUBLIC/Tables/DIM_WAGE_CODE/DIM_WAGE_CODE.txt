create or replace TABLE DIM_WAGE_CODE (
	WAGE_ID NUMBER(38,0) NOT NULL autoincrement start 1 increment 1 order,
	WAGE_CD VARCHAR(10),
	WAGE_CD_DESC_TXT VARCHAR(100),
	CREATED_DT TIMESTAMP_NTZ(0),
	CREATED_BY VARCHAR(50),
	MODIFIED_DT TIMESTAMP_NTZ(0),
	MODIFIED_BY VARCHAR(50),
	ACTIVE_STATUS VARCHAR(1)
);
